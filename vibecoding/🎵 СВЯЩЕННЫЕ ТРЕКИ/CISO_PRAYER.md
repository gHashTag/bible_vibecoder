    regulation_id,
    ai_track_regulation_changes(regulation_id, 'daily') as change_alerts,
    ai_assess_compliance_impact(regulation_id, organization_profile) as impact_score,
    ai_calculate_implementation_effort(regulation_id, current_controls) as effort_estimate,

  FROM regulatory_frameworks
  WHERE status = 'active'
),

ai_compliance_gaps AS (
  SELECT 
    control_id,
    ai_identify_control_gaps(control_id, current_implementation) as gap_analysis,
    ai_estimate_remediation_cost(control_id, gap_analysis) as remediation_cost,
    ai_prioritize_control_implementation(control_id, risk_appetite) as priority_score
  FROM security_controls
),

ai_audit_intelligence AS (
  SELECT 
    audit_framework,
    ai_predict_audit_questions(audit_framework, organization_type) as expected_questions,
    ai_generate_evidence_requirements(audit_framework) as evidence_checklist,
    ai_assess_audit_readiness(audit_framework, current_controls) as readiness_score
  FROM audit_frameworks
)

SELECT 
  'Compliance Dashboard' as report_type,
  ai_generate_compliance_executive_summary(
    rt.impact_score,
    cg.gap_analysis,
    ai_r.readiness_score
  ) as executive_summary
FROM ai_regulatory_tracking rt
JOIN ai_compliance_gaps cg ON 1=1
JOIN ai_audit_intelligence ai_r ON 1=1;
